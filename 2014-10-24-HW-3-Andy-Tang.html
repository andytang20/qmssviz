asia2006 <- readsurvey("AsiaBarometer2006.txt")
asia2007 <- readsurvey("AsiaBarometer2007.txt")

selectvariable <- function(dataset, question, year) {
    column <- match(question, colnames(dataset))
    question <- dataset[dataset[, column] != "#NULL!", column]

    number <- rep(year, length(question))
    question <- data.frame(cbind(question, number))
    colnames(question) <- c("response", "wave")
    return(question)
}

question18asia2006 <- selectvariable(asia2006, "Q18", 2006)
question19asia2007 <- selectvariable(asia2007, "Q19", 2007)

mergesurvey <- function(survey1, survey2) {
    # survey1: the first data.frame. survey2: the second data.frame.
    if (dim(survey1)[2] == dim(survey2)[2]) {
        merge <- rbind(survey1, survey2)
        return(merge)
    }
}
